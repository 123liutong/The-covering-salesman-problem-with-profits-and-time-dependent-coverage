from pathlib import Path

def read_instance_with_ct(path: str):
    lines = Path(path).read_text(encoding="utf-8").splitlines()

    # header
    h = lines[0].strip().split()
    T_max = float(h[0])
    st_max = float(h[1]) if len(h) > 1 else None

    # node lines: must be exactly 3 numeric tokens
    nodes_raw = []
    i = 1
    while i < len(lines):
        s = lines[i].strip()
        if not s:
            i += 1
            continue
        parts = s.split()
        ok = len(parts) == 3
        if ok:
            try:
                x, y, p = float(parts[0]), float(parts[1]), float(parts[2])
            except ValueError:
                ok = False
        if not ok:
            break
        nodes_raw.append((x, y, p))
        i += 1

    n = len(nodes_raw)

    # skip blank lines before CT_L
    while i < len(lines) and not lines[i].strip():
        i += 1

    # CT_L: next n lines
    CT_L = []
    for _ in range(n):
        row = [float(x) for x in lines[i].split()]
        if len(row) != n:
            raise ValueError("CT_L row length mismatch")
        CT_L.append(row)
        i += 1

    # skip blanks and read [CT_N]
    while i < len(lines) and not lines[i].strip():
        i += 1
    if i >= len(lines) or lines[i].strip() != "[CT_N]":
        raise ValueError("missing [CT_N] section")
    i += 1

    # CT_N: next n lines
    CT_N = []
    for _ in range(n):
        row = [float(x) for x in lines[i].split()]
        if len(row) != n:
            raise ValueError("CT_N row length mismatch")
        CT_N.append(row)
        i += 1

    # raw -> reader order mapping
    # raw: [start, end, customers...]
    # reader: [start, customers..., end]
    idx_reader = [0] + list(range(2, n)) + [1]
    nodes_reader = [nodes_raw[k] for k in idx_reader]

    return {
        "T_max": T_max,
        "st_max": st_max,
        "nodes_raw": nodes_raw,
        "nodes_reader": nodes_reader,
        "CT_L": CT_L,
        "CT_N": CT_N,
        "idx_reader": idx_reader,
    }
